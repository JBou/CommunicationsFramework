031public class IConnectionManager implements ConnectionManager {
032
033    public final List<IConnection> connections = Collections.synchronizedList(new ArrayList<IConnection>());
034    private final Map<Class<? extends Packet>, List<PacketListener>> listeners = new ConcurrentHashMap<>();
035    private final Map<HookType, List<Callback<Connection>>> hooks = new ConcurrentHashMap<>();
036    public final ClassLoader classLoader;
037
038    /**
039     * @param classLoader The class loader (can be gotten with getClass#getClassLoader)

042    public IConnectionManager(ClassLoader classLoader){
043        this.classLoader = classLoader;
044    }

066    @Override
067    public void registerListener(PacketListener packetListener){
068        for (Method method : packetListener.getClass().getMethods()){
069            if (isMethodApplicable(method)){
070                Class<? extends Packet> parameter = (Class<? extends Packet>) (method.getParameterCount() == 2 ? method.getParameterTypes()[1] : method.getParameterTypes()[0]);
071                List<PacketListener> list = listeners.getOrDefault(parameter, new ArrayList<PacketListener>());
072                list.add(packetListener);
073                listeners.put(parameter, list);
074            } else if (method.isAnnotationPresent(PacketHandler.class)){
075                throw new IllegalArgumentException("Method " + packetListener.getClass() + "#" + method.getName() + " has @PacketHandler annotation but does not take the correct parameters");
076            }
077        }
078    }
079
080    @Override
081    public boolean isListenerRegistered(PacketListener packetListener){
082        for (Class<? extends Packet> clazz : listeners.keySet()){
083            List<PacketListener> list = listeners.get(clazz);
084
085            if (list.contains(packetListener)){
086                return true;
087            }
088        }
089
090        return false;
091    }
092
093    @Override
094    public void unregisterListener(PacketListener packetListener){
095        for (Class<? extends Packet> clazz : listeners.keySet()){
096            List<PacketListener> list = listeners.get(clazz);
097            list.remove(packetListener);
098            listeners.put(clazz, list);
099        }
100    }
101
102    public void callListeners(IConnection connection, Packet packet){
103        if (listeners.containsKey(packet.getClass())){
104            for (PacketListener packetListener : listeners.get(packet.getClass())){
105                for (Method method : packetListener.getClass().getMethods()){
106                    if (isMethodApplicable(packet, method)){
107                        try {
108                            if (method.getParameterCount() == 2){
109                                method.invoke(packetListener, connection, packet);
110                            } else {
111                                method.invoke(packetListener, packet);
112                            }
113                        } catch (Throwable e) {
114                            System.out.println("Error whilst invoking listener\nPacket: " + packet.getClass().getName() + "\nMethod: " + packetListener.getClass().getName() + "#" + method.getName());
115                            e.printStackTrace();
116                        }
117                    }
118                }
119            }
120        }
121    }
122
123    private boolean isMethodApplicable(Packet packet, Method method){
124        return isMethodApplicable(packet.getClass(), method);
125    }
126
127    private boolean isMethodApplicable(Class<? extends Packet> packet, Method method){
128        Class<?>[] parameters = method.getParameterTypes();
129        return method.isAnnotationPresent(PacketHandler.class) && ((parameters.length == 1 && packet.isAssignableFrom(parameters[0])) || (parameters.length == 2 && parameters[0].equals(Connection.class) && packet.isAssignableFrom(parameters[1])));
130    }
131
132    private boolean isMethodApplicable(Method method){
133        Class<?>[] parameters = method.getParameterTypes();
134        return method.isAnnotationPresent(PacketHandler.class) && ((parameters.length == 1 && Packet.class.isAssignableFrom(parameters[0])) || (parameters.length == 2 && parameters[0].equals(Connection.class) && Packet.class.isAssignableFrom(parameters[1])));
135    }
136
137    @Override
138    public void addHook(HookType hookType, Callback<Connection> consumer){
139        List<Callback<Connection>> list = hooks.getOrDefault(hookType, new ArrayList<Callback<Connection>>());
140        list.add(consumer);
141        hooks.put(hookType, list);
142    }
143
144    public void callHook(IConnection connection, HookType hookType){
145        if (hooks.containsKey(hookType)){
146            for (Callback<Connection> consumer : hooks.get(hookType)){
147                consumer.call(connection);
148            }
149        }
150    }
151
152    @Override
153    public List<Connection> getConnections(){
154        return new ArrayList<Connection>(connections); // todo cache
155    }
156
157    @Override
158    public void close() throws IOException {
159        System.out.println("Closing all connections...");
160
161        for (IConnection connection : new ArrayList<>(connections)){ // create copy so we don't get Concurrent issues
162            connection.close(false);
163        }
164    }
