001package com.ikeirnez.communicationsframework.api.connection;
002
003import com.ikeirnez.communicationsframework.api.packets.Packet;
004
005import java.io.Closeable;
006import java.net.InetSocketAddress;
007
008/**
009 * Represents a connection.
010 *
011 * @author iKeirNez
012 */
013public interface Connection extends Closeable {
014
015    /**
016     * Gets the connected state of this connection
017     *
018     * @return The connected state
019     */
020    public boolean isConnected();
021
022    /**
023     * Gets the address that this Connection will connect to (hostname may be null in some ServerConnection cases)
024     *
025     * @return The address that this Connection will connect to/listen for
026     */
027    public InetSocketAddress getSocketAddress();
028
029    /**
030     * Gets the hostname that this Connection will connect to (may be null in some ServerConnection instances)
031     *
032     * @return The address that this Connection will connect to
033     */
034    public String getHostName();
035
036    /**
037     * Gets the port this Connection will use for the connection
038     *
039     * @return The port
040     */
041    public int getPort();
042
043    /**
044     * Sends a packet via this connection to the other side
045     *
046     * @param packet              The packet instance to send
047     * @param queueIfNotConnected If this connection is not currently connected (lost connection or initialising connection) defaults to true
048     */
049    public void sendPacket(Packet packet, boolean queueIfNotConnected);
050
051    /**
052     * Sends a packet via this connection to the other side
053     *
054     * @param packet The packet instance to send
055     */
056    public void sendPacket(Packet packet);
057
058}
